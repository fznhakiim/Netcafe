   public String authenticate(String username, String password) {
    if (username.isEmpty() || password.isEmpty()) {
        return "Please Enter Your Username and Password";
    }

    String Query = "SELECT * FROM `tb_username` WHERE Username = ? AND Password = ?";
    try {
        Con = DriverManager.getConnection("jdbc:mysql://localhost:3306/warned_db", "root", "");
        pst = Con.prepareStatement(Query);
        pst.setString(1, username);
        pst.setString(2, password);
        Rs = pst.executeQuery();

        if (Rs.next()) {
            String adminStatus = Rs.getString("Status");

            if ("admin super".equals(adminStatus) || "admin regular".equals(adminStatus)) {
                return "SUCCESS:" + adminStatus;
            } else {
                return "Unknown admin status";
            }
        } else {
            return "Wrong Username Or Password";
        }
    } catch (Exception e) {
        return "Error: " + e.getMessage();
    } finally {
        try {
            if (Rs != null) Rs.close();
            if (pst != null) pst.close();
            if (Con != null) Con.close();
        } catch (Exception e) {
            // Log error jika diperlukan
        }
    }
}

    
   void LoginFieldMousePressed(java.awt.event.MouseEvent evt) {                                       
     String username = UsernameField.getText();
    String password = new String(Passwordfield.getPassword());

    String authResult = authenticate(username, password);

    if (authResult.startsWith("SUCCESS")) {
        String status = authResult.split(":")[1];
        new AfterLoginGui(username, password).setVisible(true);
        this.dispose();
    } else {
        JOptionPane.showMessageDialog(this, authResult);
    }
    // Database connection and authentication logic
    try {
        Connection con = DriverManager.getConnection("jdbc:mysql://localhost:3306/ncafe", "root", "password");
        String query = "SELECT * FROM users WHERE username=? AND password=?";
        PreparedStatement pst = con.prepareStatement(query);
        pst.setString(1, username);
        pst.setString(2, password);
        ResultSet rs = pst.executeQuery();

        if (rs.next()) {
            JOptionPane.showMessageDialog(this, "Login successful!");
            // Redirect to another page or window
        } else {
            JOptionPane.showMessageDialog(this, "Invalid username or password.");
        }
        con.close();
    } catch (Exception e) {
        JOptionPane.showMessageDialog(this, "Error: " + e.getMessage());
    }
}
