    void AdminmanageButtonMouseClicked(java.awt.event.MouseEvent evt) {

        if (loggedInUsername != null && !loggedInUsername.isEmpty()) {
            String adminType = getAdminType(loggedInUsername);

            if ("admin regular".equals(adminType)) {
                new RegularAdminGUI(loggedInUsername, Password).setVisible(true);
            } else if ("admin super".equals(adminType)) {
                new SuperAdminGUI(loggedInUsername, Password).setVisible(true);
            } else {
                JOptionPane.showMessageDialog(this, "Unknown admin type for logged in user");
            }

            this.dispose();
        } else {
            JOptionPane.showMessageDialog(this, "No admin logged in");
        }
    }
    String getAdminType(String username) {
        String adminType = "";

        try {
            Connection connection = DriverManager.getConnection("jdbc:mysql://localhost:3306/warned_db", "root", "");
            Statement statement = connection.createStatement();
            ResultSet resultSet = statement.executeQuery("SELECT Status FROM tb_username WHERE Username = '" + username + "'");

            if (resultSet.next()) {
                adminType = resultSet.getString("Status");
            }
            resultSet.close();
            statement.close();
            connection.close();
        } catch (SQLException e) {
            e.printStackTrace();
        }
        return adminType;

    }


 public boolean exitConfirmed = false; // Flag untuk melacak apakah exit dikonfirmasi

void ExitButtonMouseClicked(java.awt.event.MouseEvent evt) {
    int choice;

    if (evt != null) {
        // Behavior saat runtime
        choice = JOptionPane.showConfirmDialog(this, "Are you sure you want to exit?", "Exit Confirmation", JOptionPane.YES_NO_OPTION);
    } else {
        // Behavior saat testing
        choice = JOptionPane.YES_OPTION;
    }

    if (choice == JOptionPane.YES_OPTION) {
        exitConfirmed = true;  // Flag diatur menjadi true saat exit dikonfirmasi

        // Dispose jendela saat ini (AfterLoginGui)
        this.dispose();

        // Membuka kembali LoginGUI
        java.awt.EventQueue.invokeLater(new Runnable() {
            public void run() {
                new LoginGUI().setVisible(true);  // Mengarahkan user ke LoginGUI
            }
        });
    }
}
